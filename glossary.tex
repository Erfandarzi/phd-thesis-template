\newacronym{FDD}{FDD}{Feedback-Driven Development, a model of the modern code creation cycle

  production code, originally proposed by Beck~\cite{Beck:book}}

\newacronym{GitHub}{GitHub}{The most widely used collaborative software building platform (2018), free
  to use for OSS}
\newacronym{GHTorrent}{GHTorrent}{A scalable, queriable, offline mirror of data offered

  carried out}
